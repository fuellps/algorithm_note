1.静态链表用idx表示节点编号.主要有两个操作.插入和删除.

```c++
#include <bits/stdc++.h>
using namespace std;

const int N = 100010;
//head 表示头指针,指向头节点的编号
//e[i]表示编号为idx的节点的值
//ne[i]表示编号为idx的节点的下一个节点的编号
//idx表示节点的编号.每次新建1个节点就会+1
int head = -1,e[N],ne[N],idx;
void add_to_head(int x){
    e[idx] = x;//新建节点,节点编号在下面就会加1
    ne[idx] = head;//头插法
    head = idx++;
}
void insert(int k,int x){
    e[idx] = x;//新建节点
    ne[idx] = ne[k];
    ne[k] = idx++;
}
void rm(int k){
    if(k == -1) head = ne[head];//删除头节点
    else ne[k] = ne[ne[k]];//删除第k个节点
}
void print(){
    //从头节点编号开始,遍历链表
    for(int ei = head; ei != -1; ei = ne[ei]){
        cout <<e[ei] <<" ";
    }
}

int main()
{
    int n,x,k;cin>>n;
    while (n -- ){
        string op;cin>>op;
        if(op == "H"){
            cin >>x;
            add_to_head(x);
        }else if(op == "I"){
            cin >> k >> x;
            insert(k-1,x);
        }else{
            cin >> k;
            rm(k-1);
        }
    }
    print();
    return 0;

```
